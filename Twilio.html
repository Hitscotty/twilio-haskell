type Twilio = TwilioT IO

Parse an account SID and authentication token before running zero or more REST API requests to Twilio.

For example, you can fetch the Calls resource in the IO monad as follows: 

module Main where

import Control.Monad.IO.Class (liftIO)
import System.Environment (getEnv)
import Twilio.Calls as Calls
import Twilio.Types

-- | Print calls.
main :: IO ()
main = runTwilio' (getEnv "ACCOUNT_SID")
                  (getEnv "AUTH_TOKEN")
     $ Calls.get >>= liftIO . print
